```python
class TerminalChrome:
    """Terminal Core chrome renderer with 256-color ANSI support."""

    # Palette
    AMBER = "\033[38;5;208m"
    CYAN = "\033[38;5;81m"
    GREEN = "\033[38;5;78m"
    RED = "\033[38;5;160m"
    DIM = "\033[38;5;245m"
    WHITE = "\033[38;5;15m"
    RESET = "\033[0m"
    BG_BLACK = "\033[40m"

    # Border sets
    ROUNDED = {"tl": "в•ӯ", "tr": "в•®", "bl": "в•°", "br": "в•Ҝ", "h": "в”Җ", "v": "в”Ӯ"}
    SHARP = {"tl": "в”Ң", "tr": "в”җ", "bl": "в””", "br": "в”ҳ", "h": "в”Җ", "v": "в”Ӯ"}
    DOUBLE = {"tl": "в•”", "tr": "в•—", "bl": "в•ҡ", "br": "в•қ", "h": "в•җ", "v": "в•‘"}
    ASCII = {"tl": "+", "tr": "+", "bl": "+", "br": "+", "h": "-", "v": "|"}

    def __init__(self, no_color: bool = False, no_unicode: bool = False):
        self.no_color = no_color or os.environ.get("NO_COLOR")
        self.no_unicode = no_unicode
        self.borders = self.ASCII if no_unicode else self.ROUNDED

    def _c(self, code: str, text: str) -> str:
        if self.no_color:
            return text
        return f"{code}{text}{self.RESET}"

    def border(self, style: str = "rounded") -> dict:
        if self.no_unicode:
            return self.ASCII
        return {"rounded": self.ROUNDED, "sharp": self.SHARP, "double": self.DOUBLE}.get(style, self.ROUNDED)

    def box(self, title: str, lines: list[str], width: int = 40, style: str = "rounded", color: str = None) -> str:
        b = self.border(style)
        c = color or self.CYAN
        out = []
        out.append(f"{c}{b['tl']}{b['h'] * (width - 2)}{b['tr']}{self.RESET}")
        if title:
            pad = width - len(title) - 4
            out.append(f"{c}{b['v']}{self.RESET} {self._c(self.WHITE, title)}{' ' * pad} {c}{b['v']}{self.RESET}")
            out.append(f"{c}{b['v']}{b['h'] * (width - 2)}{b['v']}{self.RESET}")
        for line in lines:
            pad = width - len(line) - 4
            out.append(f"{c}{b['v']}{self.RESET} {line}{' ' * pad} {c}{b['v']}{self.RESET}")
        out.append(f"{c}{b['bl']}{b['h'] * (width - 2)}{b['br']}{self.RESET}")
        return "\n".join(out)
```
